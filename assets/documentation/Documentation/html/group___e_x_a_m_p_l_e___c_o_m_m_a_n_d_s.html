FPS

The FPS command draws the FPS on screen.

// FPSCommand.h
#include "ConsoleCommand.h"
class DrawableFPS; // Forward deceleration of a GameObject
class DrawFPSCommand : public ConsoleCommand {
public:
DrawFPSCommand();
DrawFPSCommand(const DrawFPSCommand&) = delete;
DrawFPSCommand(DrawFPSCommand&&) = delete;
DrawFPSCommand& operator=(const DrawFPSCommand&) = delete;
DrawFPSCommand& operator=(DrawFPSCommand&&) = delete;
virtual ~DrawFPSCommand();
virtual void execute() override;
private:
DrawableFPS* obj;
bool active;
};
// FPSCommand.cpp
DrawFPSCommand::DrawFPSCommand()
: obj(new DrawableFPS), active(false)
{
}
void DrawFPSCommand::execute() {
if (active) {
active = false;
obj->SceneExit();
}
else {
active = true;
obj->SceneEntry(); // Scene entry calls draw upon the stored screen logger
}
}
DrawFPSCommand::~DrawFPSCommand() {
delete obj;
}

Map

The Map command performs a scene change upon the stored scene in a "Source Engine - esk" style.

// MapCommand.h
#include "ConsoleCommand.h"
#include <list>
class Scene;
class MapCommand : public ConsoleCommand {
public:
MapCommand();
MapCommand(const MapCommand&) = delete;
MapCommand(MapCommand&&) = delete;
MapCommand& operator=(const MapCommand&) = delete;
MapCommand& operator=(MapCommand&&) = delete;
virtual ~MapCommand();
MapCommand(Scene* newScene);
virtual void execute() override;
typedef std::list<MapCommand*> ItemList;
typedef ItemList::iterator StorageRef;
Scene* getScene();
void Initialize(Scene* nextScene, StorageRef ref);
StorageRef getStorageRef();
private:
StorageRef myStorageRef;
Scene* nextScene;
bool fired;
};
// MapCommand.cpp
MapCommand::MapCommand()
: nextScene(nullptr), fired(false)
{
}
MapCommand::MapCommand(Scene* newScene) {
nextScene = newScene;
fired = false;
}
void MapCommand::execute() {
fired = true;
SceneManager::SetNextScene(nextScene);
}
Scene* MapCommand::getScene() {
return nextScene;
}
MapCommand::StorageRef MapCommand::getStorageRef() {
return this->myStorageRef;
}
void MapCommand::Initialize(Scene* iNextScene, MapCommand::StorageRef ref) {
nextScene = iNextScene;
myStorageRef = ref;
}
MapCommand::~MapCommand() {
// We ensure the command never fired because as soon as a scene is
// "given" to the SceneManager, it is now the scene manager's
// responsibility of deleting it. Not the command's. That is
//unless, the command never triggered
if (!fired) {
delete nextScene;
}
}

Night Command

The night command simulates the changing of day. What is actually does is swaps two skyboxes and (if provided) a terrain.

// NightCommand.h
#include "ConsoleCommand.h"
class SkyBox;
class TerrainObject;
class NightCommand : public ConsoleCommand {
public:
NightCommand() = default;
NightCommand(const NightCommand&) = delete;
NightCommand(NightCommand&&) = delete;
NightCommand& operator=(const NightCommand&) = delete;
NightCommand& operator=(NightCommand&&) = delete;
virtual ~NightCommand();
NightCommand(SkyBox* daySkybox, SkyBox* nightSkybox, TerrainObject* daySkyBox = nullptr, TerrainObject* nightSkyBox = nullptr);
virtual void execute() override;
private:
SkyBox* daySkybox;
SkyBox* nightSkybox;
TerrainObject* dayTerrain;
TerrainObject* nightTerrain;
bool triggered;
};
// NightCommand.cpp
NightCommand::NightCommand(SkyBox* daySky, SkyBox* nightSky, TerrainObject* _dayTerrain, TerrainObject* _nightTerrain)
: daySkybox(daySky), nightSkybox(nightSky), dayTerrain(_dayTerrain), nightTerrain(_nightTerrain), triggered(false)
{
}
void NightCommand::execute() {
if (!triggered) { // Check to see if we've triggered this command before (to avoid double registration)
daySkybox->TerminateDraw();
nightSkybox->SetupDraw();
triggered = true;
if (dayTerrain && nightTerrain) { // If a terrain is defined, swap it too.
TerrainManager::DeinitializeCurrentTerrain();
TerrainManager::SetCurrentTerrain(nightTerrain);
nightTerrain->Initialize();
}
ConsoleManager::Print("Set the time to day", false); // Let the dev know what state it's in
}
else {
nightSkybox->TerminateDraw(); // Revert
daySkybox->SetupDraw();
triggered = false;
if (dayTerrain && nightTerrain) {
TerrainManager::DeinitializeCurrentTerrain();
TerrainManager::SetCurrentTerrain(dayTerrain);
dayTerrain->Initialize();
}
ConsoleManager::Print("Set the time to night", false);
}
}
NightCommand::~NightCommand() {
if (triggered) {
nightSkybox->TerminateDraw(); // Only required to deinitialize the draw for the night sky box if we fired it here
}
}
