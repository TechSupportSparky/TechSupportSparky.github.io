This tutorial will go over the process for creating a console object.

To begin making a console, you must first create a header file. Name the console however you would like and define the console as such:

#include "GameObject.h"
#include "Console.h"
class SlySprite; // Forward deceleration
class RadConsole : public Console, public GameObject {
public:
// Thou who forgets the Big Six, forgets themselves.
RadConsole();
RadConsole(const RadConsole&) = delete;
RadConsole(RadConsole&&) = delete;
RadConsole& operator=(const RadConsole&) = delete;
RadConsole& operator=(RadConsole&&) = delete;
virtual ~RadConsole(); // Virtual inheritance! YA!
// I force you to define an update and draw to avoid any
// "why does my code not work :( messages.
virtual void Update() {};
virtual void Draw() {};
virtual void Draw2D() override; // Not required either unless we have a background
// Handler for keyboard input
virtual void KeyPressed(AZUL_KEY k) override;
virtual void KeyReleased(AZUL_KEY k) override;
private:
// Key setup helper function
void KeyInitialize() override;
SlySprite* background
bool consoleToggled;
int asciiOffset;
}

Now with a header file defined, we jump to an implementation file (.cpp for the new users)

#include "ConsoleManager.h"
#include "RadConsole.h"
#include "SlySprite.h"
#include "Inputable.h"
#include "Console.h"
RadConsole::RadConsole() {
consoleToggled = false; // Start with the console off
asciiOffset = 32;
KeyInitialize(); // Submit inputable requests for the keyboard
// Built-in commands request
// If you do not want your user to be
// able to use these commands, leave
// them off
Reload();
GodCam();
Ping();
Time();
// User commands, these are the ones you create!
// These commands use the inherited "addCommand()"
// function.
addCommand("spawn", new SpawnTankCommand(), 1 "tank");
addCommand("spawn", new SpawnFrigateCommand(), 1, "frigate");
addCommand("spawn", new SpawnHydraCommand(), 2, "tank frigate");
// Background is not required, but it looks pretty.
// This is merely a sprite that overlays the back
// of the console.
background = new SlySprite("ConsoleBackground");
background->SetPosition(500, 500);
// Submitting a draw request for the Draw2D() function
// for the background. If you choose to not have a
// background, there is no reason for the console to draw
Drawable::SubmitDrawRegistration();
ConsoleManager::SetCurrentConsole(this); // Tell the console manager we want to use this console
}
void RadConsole::KeyInitialize() {
// LETTERS
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_A, KEY_PRESS);
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_B, KEY_PRESS);
...
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_Z, KEY_PRESS);
// NUMS
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_0, KEY_PRESS);
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_1, KEY_PRESS);
...
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_9, KEY_PRESS);
// SPECIAL
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_ENTER, KEY_PRESS); // Execute command
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_BACKSPACE, KEY_PRESS); // Backspace command line
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_SPACE, KEY_PRESS); // Space command line
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_GRAVE_ACCENT, KEY_PRESS); // Toggle console
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_LEFT_SHIFT, KEY_PRESS); // Shift letter
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_LEFT_SHIFT, KEY_RELEASE);
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_RIGHT_SHIFT, KEY_PRESS);
Inputable::SubmitKeyRegistration(AZUL_KEY::KEY_RIGHT_SHIFT, KEY_RELEASE);
}
void RadConsole::KeyPressed(AZUL_KEY k) {
if (k == AZUL_KEY::KEY_GRAVE_ACCENT) {
consoleToggled = !consoleToggled;
ConsoleLog::ToggleActive(); // Tell the logger to stop drawing...
}
else {
if (consoleToggled) {
switch (k) {
case AZUL_KEY::KEY_LEFT_SHIFT:
case AZUL_KEY::KEY_RIGHT_SHIFT:
asciiOffset = 0;
break;
case AZUL_KEY::KEY_ENTER:
this->passIntro(); // Pass intro is the execution command, KEY_ENTER is standard but not required
ConsoleLog::ClearInput();
break;
case AZUL_KEY::KEY_BACKSPACE:
this->passBackspace();
ConsoleLog::BackspaceInput();
break;
case AZUL_KEY::KEY_0:
case AZUL_KEY::KEY_1:
case AZUL_KEY::KEY_2:
case AZUL_KEY::KEY_3:
case AZUL_KEY::KEY_4:
case AZUL_KEY::KEY_5:
case AZUL_KEY::KEY_6:
case AZUL_KEY::KEY_7:
case AZUL_KEY::KEY_8:
case AZUL_KEY::KEY_9:
case AZUL_KEY::KEY_SPACE:
// avoid offsetting
this->passKey(char(k));
break;
default:
// only works for letters of course...
this->passKey(char(int(k) + asciiOffset));
}
}
}
}
void RadConsole::KeyReleased(AZUL_KEY k) {
if (k == AZUL_KEY::KEY_LEFT_SHIFT || k == AZUL_KEY::KEY_RIGHT_SHIFT) {
asciiOffset = 32;
}
}
void RadConsole::~RadConsole {
delete background;
}
void RadConsole::Draw2D() {
if (consoleToggled) {
background->Render();
}
}

That's it! The console is registered with the ConsoleManager and is ready to go. The best place to start is to get the PingCommand working first. Once that's setup, try bringing in the rest of the built-ins, then finally the developer made commands.
